import os

file_path = os.chdir('C:\ICL 2022\week3\DATA\Q1. finding_files')

all_name = os.listdir(file_path)

file_name = []

for i in all_name:
        if '.' in i:
                file_name.append(i)

print(', '.join(file_name))
